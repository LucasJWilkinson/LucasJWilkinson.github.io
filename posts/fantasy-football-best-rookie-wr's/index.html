Code 
xg_mod <- boost_tree(
  trees = tune(), 
  learn_rate = tune(),
  tree_depth = tune(), 
  min_n = tune(),
  loss_reduction = tune(), 
  sample_size = tune(), 
  mtry = tune(), 
  ) %>% 
  set_mode("regression") %>% 
  set_engine("xgboost")

lightgbm_mod <- 
  boost_tree() %>%
  set_engine("lightgbm") %>%
  set_mode(mode = "regression")

xg_workflow <- 
  workflow() %>% 
  add_model(xg_mod) %>% 
  add_recipe(player_recipe)

lightgbm_workflow <- 
  workflow() %>% 
  add_model(lightgbm_mod) %>% 
  add_recipe(player_recipe)

xgb_grid <- grid_latin_hypercube(
  trees(),
  tree_depth(),
  min_n(),
  loss_reduction(),
  sample_size = sample_prop(),
  finalize(mtry(), player_train),
  learn_rate()
)

player_folds <- vfold_cv(player_train)

xgb_res <- tune_grid(
  xg_workflow,
  resamples = player_folds,
  grid = xgb_grid,
  control = control_grid(save_pred = TRUE)
)

best_rmse <- select_best(xgb_res, "rmse")

final_xgb <- finalize_workflow(
  xg_workflow,
  best_rmse
)

xg_fit <- 
  final_xgb %>% 
  fit(data = player_train)

lightgbm_fit <- 
  lightgbm_workflow %>% 
  fit(data = player_train)

xgpredict <- 
  xg_fit %>% 
  predict(new_data = player_train) %>%
  bind_cols(player_train) 

lightgbmpredict <- 
  lightgbm_fit %>% 
  predict(new_data = player_train) %>%
  bind_cols(player_train)

currentdraft <- cfbd_draft_picks(2023) %>% filter(position == "Wide Receiver")

stats18 <- cfbd_stats_season_player(2018, category = "receiving") %>% remove_empty(which=c("cols")) %>% mutate(year = 2018)

stats19 <- cfbd_stats_season_player(2019, category = "receiving") %>% remove_empty(which=c("cols")) %>% mutate(year = 2019)

stats20 <- cfbd_stats_season_player(2020, category = "receiving") %>% remove_empty(which=c("cols")) %>% mutate(year = 2020)

stats21 <- cfbd_stats_season_player(2021, category = "receiving") %>% remove_empty(which=c("cols")) %>% mutate(year = 2021)

stats22 <- cfbd_stats_season_player(2022, category = "receiving") %>% remove_empty(which=c("cols")) %>% mutate(year = 2022)

current_stats <- bind_rows(stats18, stats19, stats20, stats21, stats22)

currentrating <- cfbd_ratings_sp(2022) %>% mutate(draft_year = year+1)

currenttotalstats <- current_stats %>% 
  group_by(athlete_id) %>%
  summarize(
    total_receptions = sum(receiving_rec),
    total_yards = sum(receiving_yds),
    total_touchdowns = sum(receiving_td),
    average_long = mean(receiving_long),
  ) %>%
  mutate(
    total_ypr = total_yards/total_receptions,
    athlete_id = as.numeric(athlete_id),
    yards_per_catch = total_yards/total_receptions

  )

currentdrafted <- currentdraft %>% inner_join(currenttotalstats, by=c("college_athlete_id"="athlete_id")) 

currentwrdraftclass <- currentdrafted %>% inner_join(currentrating, by=c("college_team"="team", "year"="draft_year"))

currentwrselected <- currentwrdraftclass %>%
  select(
   name,

    total_ypr,

currentxgpredict <-

  3. Zay Flowers A lot of experts predicted Flowers from Boston College to be the second best wide receiver in this year’s class, however the Chargers decided to draft Quentin Johnson ahead of Flowers. There is no telling if the Chargers made the right decision, however my model is backing the Chargers by predicting Flowers to finish with 85.6 fantasy points.

  4. Jordan Addison Addison was picked 23rd overall in the draft, and he was the fourth overall wide receiver picked up in the 2023 draft. The USC alumni will be lining up next to (or across the field) Justin Jefferson all year, which should help Addison receive less attention from the defensive backs. In three years at college, Addison had over 3,000 receiving yards, which helped him get picked up so early in the draft. XGBoost is predicting that Addison will score just one less point than Johnson throughout the season, finishing with 86.54 fantasy points.

Quentin Johnson Johnson was the second wide receiver drafted, and he will have the luxury of having Justin Herbert throwing to him in a pass-happy offense. My model is predicting that the TCU product will score 87.67 points next year. To put this into comparison, Brandin Cooks of the Cowboys scored 88.6 fantasy points last year in just 13 games.

If you were wondering where the hell Jaxon Smith-Njigba is, well so am I. The model is predicting him to finish with a measly 75.1 fantasy points. Another surprise from the list is how far down the 39th overall pick, Jonathan Mingo is. Mingo is predicted to finish just ahead of Smith-Njigba with 76.35 fantasy points.

If you are one to trust machine learning more than your eyes, feel free to refer to this post during your fantasy draft. I am not making any promises as to how accurate it is.
